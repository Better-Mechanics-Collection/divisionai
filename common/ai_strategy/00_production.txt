	enable = {
		is_major = yes
		threat < 0.08
		has_war = no
		OR = {
			do_expand_oil = no
			do_expand_rubber = no
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = added_military_to_civilian_factory_ratio
		value = -50
	}

		type = building_target
		id = industrial_complex
		value = 150
	}
	ai_strategy = {
		type = building_target
		id = fuel_silo
		value = -100
	}
