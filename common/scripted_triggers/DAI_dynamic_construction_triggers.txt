DAI_dc_tiny_economy = {
	OR = {
		AND = {
			is_major = no
			num_of_civilian_factories < 12
		}
		AND = {
			is_major = yes
			num_of_civilian_factories < 30
		}
	}
}

DAI_dc_low_military_base = {
	OR = {
		AND = {
			is_major = no
			num_of_civilian_factories > 14
			num_of_military_factories < 6
		}
		AND = {
			is_major = no
			num_of_civilian_factories > 22
			num_of_military_factories < 10
		}
		AND = {
			is_major = yes
			num_of_civilian_factories > 35
			num_of_military_factories < 24
		}
		AND = {
			is_major = yes
			num_of_civilian_factories > 60
			num_of_military_factories < 35
		}
	}
}

DAI_dc_military_manpower_pressure = {
	OR = {
		manpower_per_military_factory > 6000
		AND = {
			is_major = yes
			manpower_per_military_factory > 4500
		}
	}
}

DAI_dc_equipment_pressure = {
	has_equipment = { infantry_equipment < 0 }
}

DAI_dc_enemy_pressure = {
	OR = {
		enemies_strength_ratio > 0.90
		AND = {
			has_war = no
			threat > 0.15
		}
	}
}

DAI_dc_defensive_crisis = {
	has_war = yes
	OR = {
		surrender_progress > 0.12
		enemies_strength_ratio > 1.20
		has_equipment = { infantry_equipment < -5000 }
	}
}

DAI_dc_bootstrap_posture = {
	DAI_dc_tiny_economy = yes
	NOT = { DAI_dc_defensive_crisis = yes }
	OR = {
		has_war = no
		AND = {
			has_war = yes
			num_of_military_factories < 6
		}
	}
}

		check_variable = { resource@rubber < -12 }
		check_variable = { resource@oil < -12 }
	}
}

DAI_dc_allied_resources_insufficient = {
	NOT = {
		any_allied_country = {
			OR = {
				check_variable = { resource@rubber > 80 }
				check_variable = { resource@oil > 80 }
			}
		}
	}
	NOT = {
		any_subject_country = {
			OR = {
				check_variable = { resource@rubber > 40 }
				check_variable = { resource@oil > 40 }
			}
		}
	}
}

DAI_dc_can_afford_support_construction = {
	NOT = { DAI_dc_bootstrap_posture = yes }
	NOT = { has_country_flag = DAI_dc_posture_growth }
	OR = {
		AND = {
			has_country_flag = DAI_dc_profile_conservative
			OR = {
				num_of_available_civilian_factories > 24
				AND = {
					num_of_civilian_factories > 90
					num_of_available_civilian_factories > 14
				}
			}
		}
		AND = {
			NOT = { has_country_flag = DAI_dc_profile_conservative }
			NOT = { has_country_flag = DAI_dc_profile_aggressive }
			OR = {
				num_of_available_civilian_factories > 16
				AND = {
					num_of_civilian_factories > 70
					num_of_available_civilian_factories > 9
				}
			}
		}
		AND = {
			has_country_flag = DAI_dc_profile_aggressive
			OR = {
				num_of_available_civilian_factories > 10
				AND = {
					num_of_civilian_factories > 50
					num_of_available_civilian_factories > 6
				}
			}
		}
	}
}
